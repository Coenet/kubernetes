  name: mariadb-ingress
  namespace: db
  annotations:

    kubernetes.io/ingress.class: public
    nginx.ingress.kubernetes.io/rewrite-target: /

  - hosts:
    - mariadb.fandomsucks.me
    secretName: mariadb-ingress-tls
  rules:
  - host: mariadb.fandomsucks.me
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: mariadb
            port:
              number: 80
